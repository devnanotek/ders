using System;

public class Fibonacci{

  public static void Main(string[] args){
	int sayi1 = 0;
	int sayi2 = 1;
	int i,sayi3;

	Console.Write(sayi1 + " \t " + sayi2 + "\t ");
	for (i = 3; i <=10; i++){
		sayi3 = sayi1 + sayi2;
		Console.Write(sayi3 + "\t ");
	 
		sayi1 = sayi2;
		sayi2 = sayi3;
	}
			
	Console.ReadKey();
    }
}
