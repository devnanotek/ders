using System;

class CarpimTablosu{
	
     static void Main(string[] args){
		 Console.WriteLine("Çarpma Tablosu");	 
		 Console.WriteLine("-----------------");	 
		 
		 for(int a=1;a<=10;a++){ //satir
			 for(int b=1;b<=10;b++){ // sütun
				 Console.Write(b + " X " + a + " = " + (a*b));
				 Console.Write("\t\t");
			 }
			 Console.WriteLine("-----------------");
		 }
		 
	
      }
}
