                    Console.WriteLine(i);//sonuçlarý ekrana yazýyoruz

                    adet++;// yazdýðýmýz kurallara uyan sayýlarýn adeti arttýrýyoruz 
                };
               
            }
            	Console.WriteLine("Sayýlarýn toplamý : " + toplam);

        }
    }
}
